<script>

let scale = 1;

// OPEN IMAGE
function openLightbox(src){
  const modal = document.getElementById("modal");
  const img = document.getElementById("modalImg");

  modal.style.display = "flex";
  img.src = src;

  scale = 1;
  img.style.transform = "scale(1)";
}

// CLOSE MODAL
function closeLightbox(){
  document.getElementById("modal").style.display = "none";
}

// CLICK IMAGE → DO NOTHING (PREVENT CLOSE)
document.getElementById("modalImg").addEventListener("click", function(e){
  e.stopPropagation();
});

// CLICK OUTSIDE → CLOSE
document.getElementById("modal").addEventListener("click", function(e){
  if(e.target === document.getElementById("modal")){
    closeLightbox();
  }
});

// ZOOM WITH SCROLL
document.getElementById("modalImg").addEventListener("wheel", function(e){
  e.preventDefault();

  if(e.deltaY < 0){
    scale += 0.1;
  } else {
    scale -= 0.1;
    if(scale < 1) scale = 1;
  }

  this.style.transform = scale(${scale});
});

// SCROLL ANIMATION
const cards = document.querySelectorAll('.card');

window.addEventListener('scroll', () => {
  cards.forEach(card => {
    const rect = card.getBoundingClientRect().top;
    if(rect < window.innerHeight - 100){
      card.classList.add('show');
    }
  });
});

</script>
